1   package org.bremersee.ldaptive.converter;
2   
3   import static java.util.Objects.isNull;
4   
5   import org.bremersee.ldaptive.serializable.SerLdapEntry;
6   import org.ldaptive.LdapEntry;
7   import tools.jackson.core.JsonParser;
8   import tools.jackson.databind.DeserializationContext;
9   import tools.jackson.databind.deser.std.StdDeserializer;
10  
11  /**
12   * The jackson ldap entry deserializer.
13   */
14  public class JacksonLdapEntryDeserializer extends StdDeserializer<LdapEntry> {
15  
16    /**
17     * Instantiates a new jackson ldap entry deserializer.
18     */
19    public JacksonLdapEntryDeserializer() {
20      super(LdapEntry.class);
21    }
22  
23    @Override
24    public LdapEntry deserialize(
25        JsonParser jsonParser,
26        DeserializationContext deserializationContext) {
27  
28      SerLdapEntry serLdapEntry = jsonParser.readValueAs(SerLdapEntry.class);
29      return isNull(serLdapEntry) ? null : serLdapEntry.toLdapEntry();
30    }
31  }
