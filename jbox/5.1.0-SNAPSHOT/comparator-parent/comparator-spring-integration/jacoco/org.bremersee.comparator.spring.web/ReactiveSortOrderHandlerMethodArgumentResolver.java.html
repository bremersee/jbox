package org.bremersee.comparator.spring.web;

import static java.util.Objects.isNull;

import java.util.List;
import org.bremersee.comparator.model.SortOrder;
import org.springframework.core.MethodParameter;
import org.springframework.lang.NonNull;
import org.springframework.lang.Nullable;
import org.springframework.util.StringUtils;
import org.springframework.web.reactive.BindingContext;
import org.springframework.web.reactive.result.method.SyncHandlerMethodArgumentResolver;
import org.springframework.web.server.ServerWebExchange;

/**
 * The reactive sort order handler method argument resolver.
 *
 * @author Christian Bremer
 */
public class ReactiveSortOrderHandlerMethodArgumentResolver
    extends SortOrderHandlerMethodArgumentResolverSupport
    implements SyncHandlerMethodArgumentResolver {

  /**
   * Instantiates a new reactive sort order handler method argument resolver.
   */
  public ReactiveSortOrderHandlerMethodArgumentResolver() {
    super();
  }

  @Override
  public boolean supportsParameter(@NonNull MethodParameter parameter) {
    return SortOrder.class.equals(parameter.getParameterType());
  }

  @Nullable
  @Override
  public SortOrder resolveArgumentValue(
      @NonNull MethodParameter parameter,
      @NonNull BindingContext bindingContext,
      @NonNull ServerWebExchange exchange) {

    List<String> parameterValues = exchange.getRequest()
        .getQueryParams()
        .get(getParameterName(parameter));

    // No parameter
    if (isNull(parameterValues) || parameterValues.isEmpty()) {
      return getDefaultFromAnnotationOrFallback(parameter);
    }

    // Single empty parameter, e.g "sort="
    if (parameterValues.size() == 1 && !StringUtils.hasText(parameterValues.get(0))) {
      return getDefaultFromAnnotationOrFallback(parameter);
    }

    return parseParameterValues(parameterValues);
  }

}
