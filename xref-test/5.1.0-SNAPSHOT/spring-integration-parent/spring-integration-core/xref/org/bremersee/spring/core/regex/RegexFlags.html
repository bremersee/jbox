101   /**
102    * To regex flags.
103    *
104    * @return the optional
105    */
106   public Optional<Integer> toRegexFlags() {
107     int flags = 0;
108     if (isUnixLines()) {
109       flags = Pattern.UNIX_LINES;
110     }
111     if (isCaseInsensitive()) {
112       flags = flags | Pattern.CASE_INSENSITIVE;
113     }
114     if (isComments()) {
115       flags = flags | Pattern.COMMENTS;
116     }
117     if (isMultiLine()) {
118       flags = flags | Pattern.MULTILINE;
119     }
120     if (isLiteral()) {
121       flags = flags | Pattern.LITERAL;
122     }
123     if (isDotAll()) {
124       flags = flags | Pattern.DOTALL;
125     }
126     if (isUnicodeCase()) {
127       flags = flags | Pattern.UNICODE_CASE;
128     }
129     if (isCanonEq()) {
130       flags = flags | Pattern.CANON_EQ;
131     }
132     if (isUnicodeCharacterClass()) {
133       flags = flags | Pattern.UNICODE_CHARACTER_CLASS;
134     }
135     return Optional.of(flags)
136         .filter(v -> v > 0);
137   }

139   /**
140    * Compile pattern.
141    *
142    * @param regex the regex
143    * @return the pattern
144    */
145   @SuppressWarnings("MagicConstant")
146   public Pattern compile(String regex) {
147     if (isNull(regex)) {
148       return null;
149     }
150     return toRegexFlags()
151         .map(flags -> Pattern.compile(regex, flags))
152         .orElseGet(() -> Pattern.compile(regex));
153   }
