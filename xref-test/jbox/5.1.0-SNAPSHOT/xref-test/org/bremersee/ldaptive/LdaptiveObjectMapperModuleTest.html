3   import static org.assertj.core.api.Assertions.assertThat;
4   
5   import com.fasterxml.jackson.core.JsonProcessingException;
6   import com.fasterxml.jackson.databind.ObjectMapper;
7   import lombok.Data;
8   import org.junit.jupiter.api.BeforeEach;
9   import org.junit.jupiter.api.Test;
10  import org.ldaptive.dn.Dn;
11  
12  /**
13   * The ldative object mapper module test.
14   */
15  class LdaptiveObjectMapperModuleTest {
16  
17    private ObjectMapper mapper;
18  
19    private SomeLdapEntry someLdapEntry;
20  
21    private String someLdapEntryJson;
22  
23    /**
24     * Sets up.
25     */
26    @BeforeEach
27    void setUp() {
28      mapper = new ObjectMapper();
29      mapper.registerModule(new LdaptiveObjectMapperModule());
30      someLdapEntry = new SomeLdapEntry();
31      someLdapEntry.setDn(new Dn("CN=Foo,OU=Bar,dc=example,dc=com"));
32      someLdapEntry.setCn("junit");
33      someLdapEntryJson = "{\"dn\":\"CN=Foo,OU=Bar,dc=example,dc=com\",\"cn\":\"junit\"}";
34    }
35  
36    /**
37     * Deserialize.
38     *
39     * @throws JsonProcessingException the json processing exception
40     */
41    @Test
42    void deserialize() throws JsonProcessingException {
43      SomeLdapEntry actual = mapper.readValue(someLdapEntryJson, SomeLdapEntry.class);
44      assertThat(actual).isEqualTo(someLdapEntry);
45    }
46  
47    /**
48     * Serialize.
49     *
50     * @throws JsonProcessingException the json processing exception
51     */
52    @Test
53    void serialize() throws JsonProcessingException {
54      String actual = mapper.writeValueAsString(someLdapEntry);
55      assertThat(actual).isEqualTo(someLdapEntryJson);
56    }
57  
58    /**
59     * The type Some ldap entry.
60     */
61    @Data
62    static class SomeLdapEntry {
63  
64      private Dn dn;
65  
66      private String cn;
67    }
68  
69  }
