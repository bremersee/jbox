15    /**
16     * Email regex from <a href="https://emailregex.com/">emailregex.com</a> (RFC 5322 Official
17     * Standard).
18     *
19     * <p>RFC 6530 is not supported.
20     */
21    private static final String EMAIL_REGEX = "(?:[a-z0-9!#$%&'*+/=?^_`{|}~-]+"
22        + "(?:\\.[a-z0-9!#$%&'*+/=?^_`{|}~-]+)*"
23        + "|\"(?:[\\x01-\\x08\\x0b\\x0c\\x0e-\\x1f\\x21\\x23-\\x5b\\x5d-\\x7f]"
24        + "|\\\\[\\x01-\\x09\\x0b\\x0c\\x0e-\\x7f])*\")@(?:(?:[a-z0-9](?:[a-z0-9-]*[a-z0-9])?\\.)+"
25        + "[a-z0-9](?:[a-z0-9-]*[a-z0-9])?|\\[(?:(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)\\.){3}"
26        + "(?:25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?|[a-z0-9-]*[a-z0-9]:"
27        + "(?:[\\x01-\\x08\\x0b\\x0c\\x0e-\\x1f\\x21-\\x5a\\x53-\\x7f]"
28        + "|\\\\[\\x01-\\x09\\x0b\\x0c\\x0e-\\x7f])+)])";

30    /**
31     * To regex flags.
32     *
33     * @param softly the softly
34     */
35    @Test
36    void toRegexFlags(SoftAssertions softly) {
37      RegexFlags none = new RegexFlags();
38      softly.assertThat(none.toRegexFlags()).isEmpty();
39      softly.assertThat(none.toString()).doesNotContain("true");
40  
41      RegexFlags one = new RegexFlags();
42      one.setUnixLines(true);
43      softly.assertThat(one.toRegexFlags()).isPresent();
44      softly.assertThat(one.toString()).contains("true");
45  
46      RegexFlags some = one.toBuilder()
47          .caseInsensitive(true)
48          .unicodeCase(true)
49          .build();
50      softly.assertThat(some.toRegexFlags())
51          .hasValue(Pattern.UNIX_LINES | Pattern.CASE_INSENSITIVE | Pattern.UNICODE_CASE);
52      softly.assertThat(some.toString()).contains("true");
53  
54      RegexFlags all = new RegexFlags(
55          true,
56          true,
57          true,
58          true,
59          true,
60          true,
61          true,
62          true,
63          true);
64      softly.assertThat(all.toRegexFlags())
65          .hasValue(Pattern.UNIX_LINES
66              | Pattern.CASE_INSENSITIVE
67              | Pattern.COMMENTS
68              | Pattern.MULTILINE
69              | Pattern.LITERAL
70              | Pattern.DOTALL
71              | Pattern.UNICODE_CASE
72              | Pattern.CANON_EQ
73              | Pattern.UNICODE_CHARACTER_CLASS);
74      softly.assertThat(all.toString()).doesNotContain("false");
75    }

77    /**
78     * Compile.
79     *
80     * @param softly the softly
81     */
82    @Test
83    void compile(SoftAssertions softly) {
84      Pattern caseInsensitivePattern = RegexFlags.builder().caseInsensitive(true).build()
85          .compile(EMAIL_REGEX);
86      softly.assertThat(caseInsensitivePattern.matcher("FOO@example.com").matches()).isTrue();
87  
88      Pattern pattern = new RegexFlags().compile(EMAIL_REGEX);
89      softly.assertThat(pattern.matcher("FOO@example.com").matches()).isFalse();
90  
91      Pattern empty = new RegexFlags().compile(null);
92      softly.assertThat(empty).isNull();
93    }
