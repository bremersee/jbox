17  package org.bremersee.comparator;
18  
19  import java.util.Comparator;
20  import java.util.Objects;
21  import lombok.ToString;
22  import org.bremersee.comparator.model.SortOrderItem;
23  
24  /**
25   * The value comparator extracts field value of the specified field name or path and uses the
26   * specified description (ascending or descending, case-sensitive or insensitive and
27   * 'null-handling') for sorting.
28   *
29   * @author Christian Bremer
30   */
31  @ToString
32  public class ValueComparator implements Comparator<Object> {
33  
34    private final ValueExtractor valueExtractor;
35  
36    private final SortOrderItem sortOrder;
37  
38    /**
39     * Instantiates a new value comparator.
40     *
41     * @param sortOrder the sort order
42     */
43    public ValueComparator(SortOrderItem sortOrder) {
44      this(sortOrder, null);
45    }
46  
47    /**
48     * Instantiates a new value comparator.
49     *
50     * @param sortOrder the sort order
51     * @param valueExtractor a custom value extractor (if it is {@code null}, a default will be
52     *     used)
53     */
54    public ValueComparator(
55        SortOrderItem sortOrder,
56        ValueExtractor valueExtractor) {
57      this.sortOrder = Objects.requireNonNullElseGet(sortOrder, () -> SortOrderItem.by(null));
58      this.valueExtractor = valueExtractor != null ? valueExtractor : new DefaultValueExtractor();
59    }
60  
61    @Override
62    public int compare(Object o1, Object o2) {
63      final Object v1 = valueExtractor.findValue(o1, sortOrder.getField());
64      final Object v2 = valueExtractor.findValue(o2, sortOrder.getField());
65  
66      if (v1 == null && v2 == null) {
67        return 0;
68      }
69      if (v1 == null) {
70        if (sortOrder.getDirection().isAscending()) {
71          return sortOrder.getNullHandling().isNullFirst() ? -1 : 1;
72        } else {
73          return sortOrder.getNullHandling().isNullFirst() ? 1 : -1;
74        }
75      }
76      if (v2 == null) {
77        if (sortOrder.getDirection().isAscending()) {
78          return sortOrder.getNullHandling().isNullFirst() ? 1 : -1;
79        } else {
80          return sortOrder.getNullHandling().isNullFirst() ? -1 : 1;
81        }
82      }
83  
84      if (sortOrder.getDirection().isAscending() && v1 instanceof Comparable) {
85        if (sortOrder.getCaseHandling().isInsensitive()
86            && v1 instanceof String && v2 instanceof String) {
87          return ((String) v1).compareToIgnoreCase((String) v2);
88        } else {
89          //noinspection unchecked,rawtypes
90          return ((Comparable) v1).compareTo(v2);
91        }
92  
93      } else if (!sortOrder.getDirection().isAscending() && v2 instanceof Comparable) {
94  
95        if (sortOrder.getCaseHandling().isInsensitive()
96            && v1 instanceof String && v2 instanceof String) {
97          return ((String) v2).compareToIgnoreCase((String) v1);
98        } else {
99          //noinspection unchecked,rawtypes
100         return ((Comparable) v2).compareTo(v1);
101       }
102     }
103 
104     throw new ComparatorException(
105         "Comparison of field '" + sortOrder.getField() + "' is not possible.");
106   }
107 
108 }
